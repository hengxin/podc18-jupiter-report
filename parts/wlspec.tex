\begin{frame}{}
  \begin{definition}[Weak List Specification \wlspec{}~\ncite{Attiya:PODC16}]
    Informally, \wlspec{} requires the ordering between \red{elements that are not deleted} to be consistent across the system.
  \end{definition}

  \pause
  \vspace{1.00cm}
  \centerline{\blue{\large Pairwise state compatibility property:}}
  \[
    \forall \sigma, \sigma': a, b \in \sigma \cap \sigma' \implies (a \prec_{\sigma} b \iff a \prec_{\sigma'} b)
  \]

  \[
    (\sigma, \sigma': \text{ list}; \quad a, b: \text{ element}; \quad \prec_{\sigma}: \text{ precedes})
  \]

  \begin{quote}
    For any pair of list states, there cannot be two elements $a$ and $b$ such that $a$ precedes $b$ in one state
    but $b$ precedes $a$ in the other.
  \end{quote}
\end{frame}
